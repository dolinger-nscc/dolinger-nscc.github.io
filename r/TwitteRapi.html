searched.tweets <- searchTwitter(sterm, n=numT, lang="en")
searched.text <- sapply(searched.tweets, function(x) x$getText())
  1. Format the Text Data: Remove unprintables (i.e. emoticons) and create a corpus:
searched.text <- iconv(searched.text, 'UTF-8', 'ASCII') # remove smileys
searched.corpus <- Corpus(VectorSource(searched.text)) # create a corpus
  1. Create a Document Term Matrix:
term.doc.matrix <- TermDocumentMatrix(searched.corpus,
                                      control = list(removePunctuation = TRUE,
                                                     stopwords = c(sterm,"http", stopwords("english")),
                                                     removeNumbers = TRUE,tolower = TRUE))
  1. Assign the Matrix
term.doc.matrix <- as.matrix(term.doc.matrix)
  1. Establish Word Count:
word.freqs <- sort(rowSums(term.doc.matrix), decreasing=TRUE) 
df <- data.frame(word=names(word.freqs), freq=word.freqs)
  1. Create Word Cloud: rbrewer colour pallete
wordcloud(df$word, df$freq, random.order = FALSE, colors = brewer.pal(8, "Dark2"))
